def test_sort_school # rubocop:disable Metrics/MethodLength

  [
    ['Jennifer', 4], ['Kareem', 6],
    ['Christopher', 4], ['Kyle', 3]
  ].each do |name, grade|
    school.add(name, grade)
  end
  sorted = {
    3 => ['Kyle'],
    4 => %w(Christopher Jennifer),
    6 => ['Kareem']
  }
  assert_equal sorted, school.to_h
  assert_equal [3, 4, 6], school.to_h.keys
end
